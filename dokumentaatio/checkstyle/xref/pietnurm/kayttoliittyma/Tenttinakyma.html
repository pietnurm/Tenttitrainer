24  /**
25   * Luokka tenttinakyman luomiseen ja paivittamiseen.
26   * @author pieta
27   */
28  public class Tenttinakyma {
29      private Kysymysvarasto kysymysvarasto;
30      private Testi testaaKaikki;
31      private JPanel cards;
32      private JPanel tentti;
33      private JPanel tenttivalikko;
34      private JPanel mallivastaus;
35      private String omaVastaus;
36      private final static String TENTTIVALIKKO = "tenttivalikko";
37      private final static String TENTTI = "tentti";
38      private final static String MALLIVASTAUS = "mallivastaus";
39      
40      public Tenttinakyma() throws IOException {
41          this.kysymysvarasto = new Kysymysvarasto();
42          kysymysvarasto.haeTallennetutKategoriat();
43          this.omaVastaus = "";
44  //        this.testi = new Testi(kysymysvarasto);
45      }
46      
47      public JPanel luo() throws IOException {
48          testaaKaikki = new Testi(kysymysvarasto);
49          testaaKaikki.luoKysymyslista();
50          luoKomponentit();
51          
52          this.cards = new JPanel(new CardLayout());
53          cards.add(tenttivalikko, TENTTIVALIKKO);
54          cards.add(tentti, TENTTI);
55          cards.add(mallivastaus, MALLIVASTAUS);
56          
57          return cards;
58      }
59      public void luoKomponentit() {
60          this.tenttivalikko = new JPanel(new GridLayout(5, 1));
61          
62          JLabel otsikko = new JLabel("Testaa tietosi", SwingConstants.CENTER);
63          otsikko.setFont(new Font("Rockwell", Font.PLAIN, 36));
64          tenttivalikko.setBackground(new Color(0xffffff));
65          tenttivalikko.add(otsikko);
66          
67          JButton tentiKaikki = new JButton("Tenttaa kaikkia kysymyksiä");
68          tentiKaikki.setFont(new Font("Rockwell", Font.PLAIN, 20));
69          tentiKaikki.setBackground(new Color(0xffffdd));
70          tenttivalikko.add(tentiKaikki);
71          tentiKaikki.addActionListener(new ActionListener() {
72              @Override
73              public void actionPerformed(ActionEvent ae) {
74                  CardLayout cl = (CardLayout) (cards.getLayout());
75                  cl.show(cards, TENTTI);
76              }
77          });
78          
79          JButton testaa = new JButton("");
80          testaa.setFont(new Font("Rockwell", Font.PLAIN, 20));
81          testaa.setBackground(new Color(0xffffff));
82          tenttivalikko.add(testaa);
83          
84          JButton tulokset = new JButton("");
85          tulokset.setFont(new Font("Rockwell", Font.PLAIN, 20));
86          tulokset.setBackground(new Color(0xffffdd));
87          tenttivalikko.add(tulokset);
88          
89  // TENTTI        
90          
91          this.tentti = new JPanel(new GridLayout(4, 1));
92          tentti.setBackground(new Color(0xffffff));
93          JLabel otsikkoTentti = new JLabel("Tentti", SwingConstants.CENTER);
94          otsikkoTentti.setFont(new Font("Rockwell", Font.PLAIN, 36));
95          tentti.add(otsikkoTentti);
96          
97          JLabel kysymys = new JLabel(testaaKaikki.esitaKysymys(), SwingConstants.CENTER);
98          kysymys.setBackground(new Color(0xffffdd));
99          kysymys.setFont(new Font("Calibri", Font.PLAIN, 16));
100         tentti.add(kysymys);
101         
102         JTextArea vastaus = new JTextArea("VASTAUKSESI: ");
103         vastaus.setFont(new Font("Calibri", Font.PLAIN, 16));
104         vastaus.setBackground(new Color(0xffffff));
105         tentti.add(vastaus);
106         
107         JButton tarkistaVastaus = new JButton("Tarkista vastaus");
108         tarkistaVastaus.setFont(new Font("Rockwell", Font.PLAIN, 20));
109         tarkistaVastaus.setBackground(new Color(0xffffdd));
110         tarkistaVastaus.addActionListener(new ActionListener() {
111             @Override
112             public void actionPerformed(ActionEvent ae) {
113                 omaVastaus = vastaus.getText();
114                 System.out.println(vastaus.getText());
115                 CardLayout cl = (CardLayout) (cards.getLayout());
116                 cl.show(cards, MALLIVASTAUS);
117             }
118         });
119         tentti.add(tarkistaVastaus);
120         
121         this.mallivastaus = new JPanel(new GridLayout(4, 1));
122         mallivastaus.setBackground(new Color(0xffffff));
123         
124         JLabel vastauksesi = new JLabel(omaVastaus, SwingConstants.CENTER);
125         vastauksesi.setFont(new Font("Calibri", Font.PLAIN, 16));
126         mallivastaus.add(vastauksesi);
127         
128         JLabel vastauksesi2 = new JLabel("pökäle", SwingConstants.CENTER);
129         vastauksesi2.setFont(new Font("Calibri", Font.PLAIN, 16));
130         mallivastaus.add(vastauksesi2);
131              
132     }
