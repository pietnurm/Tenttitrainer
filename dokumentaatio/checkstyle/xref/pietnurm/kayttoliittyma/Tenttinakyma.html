27  /**
28   * Luokka tenttinakyman luomiseen ja paivittamiseen.
29   * @author pieta
30   */
31  public class Tenttinakyma {
32      private Kayttoliittyma kayttoliittyma;
33      private Kysymysvarasto kysymysvarasto;
34      private Testi tenttaa;
35      private JPanel cards;
36      private JPanel tentti;
37      private JPanel tenttivalikko;
38      private JPanel mallivastaus;
39      private JPanel lopetusruutu;
40      private final static String TENTTIVALIKKO = "tenttivalikko";
41      private final static String TENTTI = "tentti";
42      private final static String MALLIVASTAUS = "mallivastaus";
43      private final static String LOPETUSRUUTU = "lopetusruutu";
44      
45      public Tenttinakyma(Kayttoliittyma kayttoliittyma) throws IOException {
46          this.kayttoliittyma = kayttoliittyma;
47          this.kysymysvarasto = new Kysymysvarasto();
48          kysymysvarasto.haeTallennetutKategoriat();
49          tenttaa = new Testi(kysymysvarasto);
50          tenttaa.luoKysymyslista();
51  
52      }
53      /**
54       * Palauttaa tenttinakyman JPanelina.
55       * @return
56       * @throws IOException 
57       */
58      public JPanel luo() throws IOException {
59          luoKomponentit();
60          
61          this.cards = new JPanel(new CardLayout());
62          cards.add(tenttivalikko, TENTTIVALIKKO);
63          cards.add(tentti, TENTTI);
64          cards.add(mallivastaus, MALLIVASTAUS);
65          cards.add(lopetusruutu, LOPETUSRUUTU);
66          
67          return cards;
68      }
69      /**
70       * Luo tenttinakyman komponentit.
71       * @throws IOException 
72       */
73      public void luoKomponentit() throws IOException {
74          this.tenttivalikko = new JPanel(new GridLayout(5, 1));
75          
76          JLabel otsikko = new JLabel("Testaa tietosi", SwingConstants.CENTER);
77          otsikko.setFont(new Font("Rockwell", Font.PLAIN, 36));
78          tenttivalikko.setBackground(new Color(0xffffff));
79          tenttivalikko.add(otsikko);
80          
81          JButton tentiKaikki = new JButton("Tenttaa kaikkia kysymyksiä");
82          tentiKaikki.setFont(new Font("Rockwell", Font.PLAIN, 20));
83          tentiKaikki.setBackground(new Color(0xffffdd));
84          tenttivalikko.add(tentiKaikki);
85          tentiKaikki.addActionListener(new ActionListener() {
86              @Override
87              public void actionPerformed(ActionEvent ae) {
88                  tenttaa = new Testi(kysymysvarasto);
89                  tenttaa.luoKysymyslista();
90                  CardLayout cl = (CardLayout) (cards.getLayout());
91                  cl.show(cards, TENTTI);
92              }
93          });
94          
95          JPanel keskipaneeli = new JPanel(new GridLayout(2, 1));
96          keskipaneeli.setBackground(new Color(0xffffff));
97          
98          Kategoriavalitsin kategoriavalitsin = new Kategoriavalitsin();
99          kategoriavalitsin.initPelkistetty();
100         JPanel valitsin = kategoriavalitsin.palautaValitsin();
101         
102         JButton tentiKategoria = new JButton("Tenttaa valitsemastasi kategoriasta");
103         tentiKategoria.setFont(new Font("Rockwell", Font.PLAIN, 20));
104         tentiKategoria.setBackground(new Color(0xffffff));
105         tentiKategoria.addActionListener(new ActionListener() {
106             @Override
107             public void actionPerformed(ActionEvent ae) {
108                 String kategorianNimi = kategoriavalitsin.palautaKategoria();
109                 Kategoria valittuKategoria;
110                 try {
111                     valittuKategoria = new Kategoria(kategorianNimi);
112                     valittuKategoria.haeTallennetutKysymykset();
113                     tenttaa = new Testi(kysymysvarasto, valittuKategoria);
114                     tenttaa.luoKysymyslista();
115                     CardLayout cl = (CardLayout) (cards.getLayout());
116                     cl.show(cards, TENTTI);
117                 } catch (IOException ex) {
118                     Logger.getLogger(Tenttinakyma.class.getName()).log(Level.SEVERE, null, ex);
119                 }
120             }
121         });
122         
123         keskipaneeli.add(tentiKategoria);
124         keskipaneeli.add(valitsin);   
125         tenttivalikko.add(keskipaneeli);
126         
127         JButton palaaValikkoon = new JButton("Takaisin valikkoon");
128         palaaValikkoon.setFont(new Font("Rockwell", Font.PLAIN, 20));
129         palaaValikkoon.setBackground(new Color(0xffffdd));
130         tenttivalikko.add(palaaValikkoon);
131         palaaValikkoon.addActionListener(new ActionListener() {
132             @Override
133             public void actionPerformed(ActionEvent ae) {
134                 CardLayout cl = (CardLayout) (kayttoliittyma.cards.getLayout());
135                 cl.show(kayttoliittyma.cards, kayttoliittyma.VALIKKOPANEL);
136             }
137         });
138         
139 // TENTTI        
140         
141         this.tentti = new JPanel(new GridLayout(4, 1));
142         tentti.setBackground(new Color(0xffffff));
143         JLabel otsikkoTentti = new JLabel("Tentti", SwingConstants.CENTER);
144         otsikkoTentti.setFont(new Font("Rockwell", Font.PLAIN, 36));
145         tentti.add(otsikkoTentti);
146         
147         JLabel kysymys = new JLabel("", SwingConstants.CENTER);
148         kysymys.setText(tenttaa.esitaKysymys());
149         kysymys.setBackground(new Color(0xffffdd));
150         kysymys.setFont(new Font("Calibri", Font.PLAIN, 16));
151         tentti.add(kysymys);
152         
153         // luodaan mallivastaus-cardia varten
154         JLabel kysymysUudestaan = new JLabel(tenttaa.esitaKysymys(), SwingConstants.CENTER);
155         kysymys.setBackground(new Color(0xffffdd));
156         kysymys.setFont(new Font("Calibri", Font.PLAIN, 16));
157         
158         JTextArea vastaus = new JTextArea("VASTAUKSESI: ");
159         vastaus.setFont(new Font("Calibri", Font.PLAIN, 16));
160         vastaus.setBackground(new Color(0xffffff));
161         tentti.add(vastaus);
162         
163         
164         
165         // luodaan jlabel mallivastaus-cardia varten
166         JLabel vastauksesi = new JLabel();
167         vastauksesi.setFont(new Font("Calibri", Font.PLAIN, 16));
168         
169         JLabel mallivast = new JLabel();
170         mallivast.setFont(new Font("Calibri", Font.PLAIN, 16));
171         
172         JPanel nappipaneeli = new JPanel();
173         nappipaneeli.setBackground(new Color(0xffffff));
174         tentti.add(nappipaneeli);
175         
176         JButton tarkistaVastaus = new JButton("Tarkista vastaus");
177         tarkistaVastaus.setFont(new Font("Rockwell", Font.PLAIN, 20));
178         tarkistaVastaus.setBackground(new Color(0xffffdd));
179         tarkistaVastaus.addActionListener(new ActionListener() {
180             @Override
181             public void actionPerformed(ActionEvent ae) {
182                 vastauksesi.setText(vastaus.getText());
183                 kysymysUudestaan.setText(tenttaa.esitaKysymys());
184                 mallivast.setText("<html>MALLIVASTAUS:<br>" + tenttaa.esitaMallivastaus() + "</html>");
185                 CardLayout cl = (CardLayout) (cards.getLayout());
186                 cl.show(cards, MALLIVASTAUS);
187             }
188         });
189         nappipaneeli.add(tarkistaVastaus);
190         
191         // luodaan keskiarvo-label lopetusruutua varten
192         JLabel keskiarvo = new JLabel("", SwingConstants.CENTER);
193         keskiarvo.setFont(new Font("Rockwell", Font.PLAIN, 46));
194         
195         
196         JButton lopetaTentti = new JButton("Lopeta tentti");
197         lopetaTentti.setFont(new Font("Rockwell", Font.PLAIN, 20));
198         lopetaTentti.setBackground(new Color(0xffffdd));
199         tentti.add(lopetaTentti);
200         lopetaTentti.addActionListener(new ActionListener() {
201             @Override
202             public void actionPerformed(ActionEvent ae) {
203                 keskiarvo.setText(new String(String.format("%.1f", tenttaa.testinKeskiarvo())));
204                 CardLayout cl = (CardLayout) (cards.getLayout());
205                 cl.show(cards, LOPETUSRUUTU);
206             }
207         });
208         nappipaneeli.add(lopetaTentti);
209         
210         // MALLIVASTAUS/ARVIOINTI
211         
212         this.mallivastaus = new JPanel(new GridLayout(4, 1));
213         mallivastaus.setBackground(new Color(0xffffff));
214         mallivast.setBackground(new Color(0xffffdd));
215         mallivastaus.add(kysymysUudestaan);
216         mallivastaus.add(mallivast);
217         mallivastaus.add(vastauksesi);
218         
219         JPanel nappipaneeli2 = new JPanel(new GridLayout(3, 1));
220         nappipaneeli2.setBackground(new Color(0xffffff));
221         mallivastaus.add(nappipaneeli2);
222         
223         JLabel arviointiteksti = new JLabel("Arvostele vastauksesi", SwingConstants.CENTER);
224         arviointiteksti.setBackground(new Color(0xffffdd));
225         arviointiteksti.setFont(new Font("Rockwell", Font.PLAIN, 16));
226         nappipaneeli2.add(arviointiteksti);
227         
228         JSlider arviointiSlider = new JSlider(JSlider.HORIZONTAL, 0, 5, 1);
229         arviointiSlider.setBackground(new Color(0xffffff));
230         arviointiSlider.setMajorTickSpacing(1);
231         arviointiSlider.setPaintTicks(true);
232         arviointiSlider.setPaintLabels(true);
233         arviointiSlider.setFont(new Font("Rockwell", Font.PLAIN, 16));
234         
235         nappipaneeli2.add(arviointiSlider);
236         
237         JButton seuraava = new JButton("Seuraava kysymys");
238         seuraava.setFont(new Font("Rockwell", Font.PLAIN, 20));
239         seuraava.setBackground(new Color(0xffffdd));
240         seuraava.addActionListener(new ActionListener() {
241             @Override
242             public void actionPerformed(ActionEvent ae) {
243                 CardLayout cl = (CardLayout) (cards.getLayout());
244                 if (tenttaa.esitaKysymys().equals("Testi on päättynyt.")) {
245                     try {
246                         tenttaa.arvostele(arviointiSlider.getValue());
247                     } catch (IOException ex) {
248                         Logger.getLogger(Tenttinakyma.class.getName()).log(Level.SEVERE, null, ex);
249                     }
250                     keskiarvo.setText(new String(String.format("%.1f", tenttaa.testinKeskiarvo())));
251                     cl.show(cards, LOPETUSRUUTU);
252                 } else {
253                     try {
254                         tenttaa.arvostele(arviointiSlider.getValue());
255                     } catch (IOException ex) {
256                         Logger.getLogger(Tenttinakyma.class.getName()).log(Level.SEVERE, null, ex);
257                     }
258                     kysymys.setText(tenttaa.esitaKysymys());
259                     vastaus.setText("VASTAUKSESI: ");
260                     cl.show(cards, TENTTI);
261                 }
262             }
263         });
264         nappipaneeli2.add(seuraava);  
265         
266         // lopetusruutu
267         this.lopetusruutu = new JPanel(new GridLayout(3, 1));
268         lopetusruutu.setBackground(new Color(0xffffff));
269         JLabel loppuviesti = new JLabel("Tentti on päättynyt.", SwingConstants.CENTER);
270         loppuviesti.setFont(new Font("Rockwell", Font.PLAIN, 36));
271         lopetusruutu.add(loppuviesti);
272         
273         JPanel keskiarvopaneeli = new JPanel(new GridLayout(2, 1));
274         keskiarvopaneeli.setBackground(new Color(0xffffff));
275         lopetusruutu.add(keskiarvopaneeli);
276         
277         JLabel keskiarvoOtsikko = new JLabel("Vastausten keskiarvo: ", SwingConstants.CENTER);
278         keskiarvoOtsikko.setFont(new Font("Rockwell", Font.PLAIN, 20));
279         keskiarvopaneeli.add(keskiarvoOtsikko);
280 //      keskiarvo-label on luotu aikaisemmin
281         keskiarvopaneeli.add(keskiarvo);
282         
283         JPanel nappipaneeli3 = new JPanel(new GridLayout(2, 1));
284         nappipaneeli3.setBackground(new Color(0xffffff));
285         lopetusruutu.add(nappipaneeli3);
286         
287         nappipaneeli3.add(new JLabel()); //tyhja valiin
288         
289         JButton lopeta = new JButton("Lopeta");
290         lopeta.setFont(new Font("Rockwell", Font.PLAIN, 20));
291         lopeta.setBackground(new Color(0xffffdd));
292         nappipaneeli3.add(lopeta);
293         lopeta.addActionListener(new ActionListener() {
294             @Override
295             public void actionPerformed(ActionEvent ae) {
296                 CardLayout cl = (CardLayout) (cards.getLayout());
297                 cl.show(cards, TENTTIVALIKKO);
298             }
299         });
